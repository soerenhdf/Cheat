--# Ultimate Automatic Vlad Ball Hack für Arceus X
--# Menü Version ohne Key-System

local HttpService = game:GetService("HttpService")
local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer
local Mouse = LocalPlayer:GetMouse()
local Ball = nil

-- GUI für das Menü erstellen
local screenGui = Instance.new("ScreenGui")
screenGui.Parent = LocalPlayer.PlayerGui

local mainFrame = Instance.new("Frame")
mainFrame.Size = UDim2.new(0, 300, 0, 400)
mainFrame.Position = UDim2.new(0.5, -150, 0.5, -200)
mainFrame.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
mainFrame.BackgroundTransparency = 0.5
mainFrame.Parent = screenGui

local title = Instance.new("TextLabel")
title.Size = UDim2.new(1, 0, 0, 50)
title.BackgroundTransparency = 1
title.Text = "Vlad Ball Hacks"
title.TextColor3 = Color3.fromRGB(255, 255, 255)
title.TextSize = 24
title.Font = Enum.Font.SourceSans
title.TextAlignment = Enum.TextAlignment.Center
title.Parent = mainFrame

-- Schaltflächen für die Hacks
local function createButton(text, position, func)
    local button = Instance.new("TextButton")
    button.Size = UDim2.new(1, 0, 0, 50)
    button.Position = position
    button.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
    button.Text = text
    button.TextColor3 = Color3.fromRGB(255, 255, 255)
    button.TextSize = 18
    button.Font = Enum.Font.SourceSans
    button.TextButtonMode = Enum.TextButtonMode.Scriptable
    button.Parent = mainFrame
    button.MouseButton1Click:Connect(func)
    return button
end

-- Funktionen für die Hacks
function findBall()
    for _, v in ipairs(workspace:GetChildren()) do
        if v:IsA("Part") and v.Name == "VladBall" then
            return v
        end
    end
    return nil
end

function autoCatchBall()
    Ball = findBall()
    if Ball then
        -- Teleportiere zu Ball, fange ihn
        local ballPos = Ball.Position
        LocalPlayer.Character.HumanoidRootPart.CFrame = CFrame.new(ballPos)
        print("Ball automatisch gefangen!")
    end
end

function instantScore()
    if Ball then
        -- Setzt den Ball zur Zielposition
        Ball.CFrame = CFrame.new(Vector3.new(1000, 1000, 1000))
        -- Score +10 hacken
        local score = LocalPlayer.leaderstats.Score
        score.Value = score.Value + 10
        print("Sieg sofort erzielt!")
    end
end

function superSpeed()
    game:GetService("TweenService"):Create(LocalPlayer.Character.HumanoidRootPart, TweenInfo.new(0.1, Enum.EasingStyle.Linear, Enum.EasingDirection.InOut, -1, true), {Position = Mouse.Hit.p}):Play()
    print("Super-Speed automatisch aktiviert!")
end

function ballAimbot()
    if Ball then
        local target = workspace:FindFirstChild("Enemy") -- Setzt hier die Logik für Gegner
        if target then
            Ball.CFrame = CFrame.new(target.Position) -- Ball wird immer zu Gegnern gelenkt
            print("Aimbot aktiviert!")
        end
    end
end

function noCooldown()
    -- Setzt den Ball-Wurf so, dass keine Abklingzeit existiert
    Ball.Velocity = Vector3.new(1000, 1000, 1000)
    print("Kein Cooldown Modus aktiviert!")
end

function infiniteJump()
    local bodyVelocity = Instance.new("BodyVelocity")
    bodyVelocity.MaxForce = Vector3.new(100000, 100000, 100000)
    bodyVelocity.Velocity = Vector3.new(0, 100, 0)
    bodyVelocity.Parent = LocalPlayer.Character.HumanoidRootPart
    print("Unendlich Sprünge aktiviert!")
end

function instantRespawn()
    wait(0.1)
    LocalPlayer.Character:BreakJoints()
    wait(0.1)
    LocalPlayer.Character:LoadCharacter()
    print("Sofortiges Respawn nach Wurf aktiviert!")
end

function enemyBlocker()
    for _, v in ipairs(workspace:GetChildren()) do
        if v:IsA("Part") and v.Name == "VladBall" and v.Parent.Name ~= LocalPlayer.Name then
            v.CFrame = CFrame.new(Vector3.new(9999, 9999, 9999)) -- Blockiere Gegner
            print("Gegner Blocker aktiviert!")
        end
    end
end

function teleportAura()
    for _, v in ipairs(workspace:GetChildren()) do
        if v:IsA("Part") and v.Name == "VladBall" then
            local distance = 10 -- Abstand zum Ball (kann angepasst werden)
            local ballPos = v.Position
            LocalPlayer.Character.HumanoidRootPart.CFrame = CFrame.new(ballPos + Vector3.new(distance, 0, distance))
            print("Teleport-Aura aktiviert! Fliege um den Ball!")
            break
        end
    end
end

-- Menü-Buttons
createButton("Auto-Catch Ball", UDim2.new(0, 0, 0.1, 0), autoCatchBall)
createButton("Instant Score", UDim2.new(0, 0, 0.2, 0), instantScore)
createButton("Super-Speed", UDim2.new(0, 0, 0.3, 0), superSpeed)
createButton("Ball-Aimbot", UDim2.new(0, 0, 0.4, 0), ballAimbot)
createButton("No Cooldown", UDim2.new(0, 0, 0.5, 0), noCooldown)
createButton("Infinite Jump", UDim2.new(0, 0, 0.6, 0), infiniteJump)
createButton("Instant Respawn", UDim2.new(0, 0, 0.7, 0), instantRespawn)
createButton("Enemy Blocker", UDim2.new(0, 0, 0.8, 0), enemyBlocker)
createButton("Teleport Aura", UDim2.new(0, 0, 0.9, 0), teleportAura)
