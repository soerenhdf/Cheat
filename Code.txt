-- Leeres Menü für Roblox (einfügen z.B. in StarterGui)

local player = game.Players.LocalPlayer
local playerGui = player:WaitForChild("PlayerGui")

-- Erstelle ein ScreenGui
local screenGui = Instance.new("ScreenGui")
screenGui.Name = "LeeresMenue"
screenGui.ResetOnSpawn = false
screenGui.Parent = playerGui

-- Rahmen (Hauptmenü-Box)
local frame = Instance.new("Frame")
frame.Size = UDim2.new(0, 300, 0, 200)
frame.Position = UDim2.new(0.5, -150, 0.5, -100)
frame.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
frame.BorderSizePixel = 0
frame.Parent = screenGui

-- Titel
local title = Instance.new("TextLabel")
title.Text = "Leeres Menü"
title.Size = UDim2.new(1, 0, 0, 40)
title.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
title.TextColor3 = Color3.fromRGB(255, 255, 255)
title.Font = Enum.Font.SourceSansBold
title.TextSize = 24
title.Parent = frame

-- Beispiel-Buttons
for i = 1, 3 do
    local button = Instance.new("TextButton")
    button.Size = UDim2.new(0.8, 0, 0, 30)
    button.Position = UDim2.new(0.1, 0, 0, 50 + (i - 1) * 40)
    button.BackgroundColor3 = Color3.fromRGB(80, 80, 80)
    button.Text = "Option " .. i
    button.TextColor3 = Color3.fromRGB(255, 255, 255)
    button.Font = Enum.Font.SourceSans
    button.TextSize = 20
    button.Parent = frame

    -- Platzhalterfunktion
    button.MouseButton1Click:Connect(function()
        print("Option " .. i .. " ausgewählt (noch keine Funktion)")
    end)
end
