-- Vlad Ball Ultimate Hack Menu (Arceus X)
local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer

-- Warte auf PlayerGui
repeat wait() until LocalPlayer:FindFirstChild("PlayerGui")

-- GUI erstellen
local gui = Instance.new("ScreenGui", LocalPlayer.PlayerGui)
gui.ResetOnSpawn = false

local frame = Instance.new("Frame", gui)
frame.Size = UDim2.new(0, 300, 0, 500)
frame.Position = UDim2.new(0.1, 0, 0.1, 0)
frame.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
frame.BorderSizePixel = 0

local title = Instance.new("TextLabel", frame)
title.Size = UDim2.new(1, 0, 0, 50)
title.Text = "Vlad Ball Hack Men√º"
title.TextColor3 = Color3.fromRGB(255, 255, 255)
title.BackgroundTransparency = 1
title.Font = Enum.Font.SourceSansBold
title.TextSize = 24

-- Funktion zum Erstellen von Buttons
local yOffset = 60
function createButton(text, callback)
	local button = Instance.new("TextButton", frame)
	button.Size = UDim2.new(1, -20, 0, 40)
	button.Position = UDim2.new(0, 10, 0, yOffset)
	button.Text = text
	button.TextColor3 = Color3.fromRGB(255, 255, 255)
	button.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
	button.BorderSizePixel = 0
	button.Font = Enum.Font.SourceSans
	button.TextSize = 18
	button.MouseButton1Click:Connect(callback)
	yOffset = yOffset + 45
end

-- Funktionen
function findBall()
	for _, v in ipairs(workspace:GetDescendants()) do
		if v:IsA("Part") and v.Name:lower():find("ball") then
			return v
		end
	end
end

createButton("Auto Catch Ball", function()
	local ball = findBall()
	if ball then
		LocalPlayer.Character.HumanoidRootPart.CFrame = ball.CFrame
	end
end)

createButton("Instant Score", function()
	local ball = findBall()
	if ball then
		ball.CFrame = CFrame.new(Vector3.new(9999, 9999, 9999))
	end
end)

createButton("Super Speed", function()
	LocalPlayer.Character.Humanoid.WalkSpeed = 100
end)

createButton("Reset Speed", function()
	LocalPlayer.Character.Humanoid.WalkSpeed = 16
end)

createButton("Infinite Jump", function()
	game:GetService("UserInputService").JumpRequest:Connect(function()
		LocalPlayer.Character:FindFirstChildOfClass("Humanoid"):ChangeState("Jumping")
	end)
end)

createButton("Teleport Aura", function()
	while wait(0.1) do
		local ball = findBall()
		if ball then
			local pos = ball.Position + Vector3.new(5, 0, 5)
			LocalPlayer.Character.HumanoidRootPart.CFrame = CFrame.new(pos)
		end
	end
end)

createButton("Instant Respawn", function()
	LocalPlayer.Character:BreakJoints()
end)

createButton("Enemy Blocker", function()
	for _, player in pairs(Players:GetPlayers()) do
		if player ~= LocalPlayer and player.Character then
			player.Character:BreakJoints()
		end
	end
end)

createButton("No Cooldown (Ball Speed)", function()
	local ball = findBall()
	if ball then
		ball.Velocity = Vector3.new(999, 0, 999)
	end
end)

createButton("Ball Aimbot", function()
	local ball = findBall()
	local target = nil
	for _, player in pairs(Players:GetPlayers()) do
		if player ~= LocalPlayer and player.Character then
			target = player.Character:FindFirstChild("HumanoidRootPart")
			break
		end
	end
	if ball and target then
		ball.CFrame = CFrame.new(target.Position)
	end
end)

-- Frame draggen lassen
local UIS = game:GetService("UserInputService")
local dragging, dragInput, dragStart, startPos

frame.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 then
		dragging = true
		dragStart = input.Position
		startPos = frame.Position

		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then
				dragging = false
			end
		end)
	end
end)

frame.InputChanged:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseMovement then
		dragInput = input
	end
end)

UIS.InputChanged:Connect(function(input)
	if input == dragInput and dragging then
		local delta = input.Position - dragStart
		frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
	end
end)
