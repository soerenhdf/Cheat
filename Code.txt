-- Roblox Hack Menu with Key System and Functions

-- Services
local Players = game:GetService("Players")
local HttpService = game:GetService("HttpService")
local UserInputService = game:GetService("UserInputService")
local StarterGui = game:GetService("StarterGui")

local player = Players.LocalPlayer
local mouse = player:GetMouse()

-- Key URL
local keyURL = "https://raw.githubusercontent.com/soerenhdf/Key/refs/heads/main/key.txt"
local validKey = nil
local currentKey = ""

-- GUI setup for the Key System
local function createKeyUI()
    local ScreenGui = Instance.new("ScreenGui")
    ScreenGui.Name = "KeySystem"
    ScreenGui.ResetOnSpawn = false
    ScreenGui.Parent = game.CoreGui

    local Frame = Instance.new("Frame")
    Frame.Size = UDim2.new(0, 300, 0, 200)
    Frame.Position = UDim2.new(0.5, -150, 0.5, -100)
    Frame.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
    Frame.BorderSizePixel = 0
    Frame.Name = "KeyFrame"
    Frame.Parent = ScreenGui

    local Title = Instance.new("TextLabel")
    Title.Text = "Enter Key to Access Menu"
    Title.Size = UDim2.new(1, 0, 0, 30)
    Title.BackgroundTransparency = 1
    Title.TextColor3 = Color3.fromRGB(255, 255, 255)
    Title.TextScaled = true
    Title.Parent = Frame

    local KeyBox = Instance.new("TextBox")
    KeyBox.Size = UDim2.new(1, 0, 0, 30)
    KeyBox.Position = UDim2.new(0, 0, 0.2, 0)
    KeyBox.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
    KeyBox.TextColor3 = Color3.fromRGB(255, 255, 255)
    KeyBox.PlaceholderText = "Enter Key"
    KeyBox.TextScaled = true
    KeyBox.ClearTextOnFocus = true
    KeyBox.Parent = Frame

    local SubmitButton = Instance.new("TextButton")
    SubmitButton.Size = UDim2.new(1, 0, 0, 40)
    SubmitButton.Position = UDim2.new(0, 0, 0.6, 0)
    SubmitButton.Text = "Submit"
    SubmitButton.BackgroundColor3 = Color3.fromRGB(100, 100, 255)
    SubmitButton.TextColor3 = Color3.fromRGB(255, 255, 255)
    SubmitButton.TextScaled = true
    SubmitButton.Parent = Frame

    -- Open link button
    local LinkButton = Instance.new("TextButton")
    LinkButton.Size = UDim2.new(1, 0, 0, 30)
    LinkButton.Position = UDim2.new(0, 0, 0.8, 0)
    LinkButton.Text = "This is the Key"
    LinkButton.BackgroundColor3 = Color3.fromRGB(0, 0, 255)
    LinkButton.TextColor3 = Color3.fromRGB(255, 255, 255)
    LinkButton.TextScaled = true
    LinkButton.Parent = Frame

    -- Open URL when clicked
    LinkButton.MouseButton1Click:Connect(function()
        setclipboard("https://link-target.net/1339083/key")
        print("Link copied to clipboard. Open your browser to get the key.")
    end)

    -- Check key when submitted
    SubmitButton.MouseButton1Click:Connect(function()
        currentKey = KeyBox.Text
        if currentKey == validKey then
            print("Valid Key entered!")
            -- Close the key entry GUI
            ScreenGui:Destroy()
            createHackMenu()  -- Create Hack Menu after valid key
        else
            print("Invalid Key! Try again.")
        end
    end)
end

-- Function to fetch the valid key from GitHub
local function fetchValidKey()
    local response = HttpService:GetAsync(keyURL)
    validKey = response
end
