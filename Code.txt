--[[
  Arceus X NEO - Obby KI Script
  Läuft automatisch durch Obbys, erkennt Hindernisse
--]]

local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local player = Players.LocalPlayer

repeat wait() until player.Character and player.Character:FindFirstChild("HumanoidRootPart")
local character = player.Character
local humanoid = character:WaitForChild("Humanoid")
local hrp = character:WaitForChild("HumanoidRootPart")

-- Funktion zum Hervorheben von Hindernissen (ESP)
for _, part in pairs(workspace:GetDescendants()) do
    if part:IsA("Part") and part.CanCollide and part.Position.Y > 2 then
        local box = Instance.new("BoxHandleAdornment")
        box.Adornee = part
        box.AlwaysOnTop = true
        box.ZIndex = 10
        box.Size = part.Size + Vector3.new(0.1, 0.1, 0.1)
        box.Color3 = Color3.fromRGB(255, 0, 0)
        box.Transparency = 0.4
        box.Parent = part
    end
end

-- Automatischer Vorwärtslauf
local function moveForward()
    if humanoid and humanoid.MoveToFinished then
        local pos = hrp.Position
        local forward = Vector3.new(0, 0, -5)
        humanoid:MoveTo(pos + forward)
    end
end

-- Hindernissen ausweichen (simpler check)
local function avoidObstacles()
    local ray = Ray.new(hrp.Position, Vector3.new(0, 0, -5))
    local hit = workspace:FindPartOnRay(ray, character)
    if hit and hit.CanCollide then
        -- Springen wenn Hindernis erkannt wird
        humanoid:ChangeState(Enum.HumanoidStateType.Jumping)
    end
end

-- Hauptloop
RunService.RenderStepped:Connect(function()
    avoidObstacles()
    moveForward()
end)
